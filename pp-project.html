Data Processing Pipeline

-- SQL: Advanced Feature Engineering WITH customer_metrics AS ( SELECT customer_id, COUNT(*) as transaction_count, AVG(transaction_amount) as avg_transaction, STDDEV(transaction_amount) as transaction_volatility, DATE_PART('day', MAX(transaction_date) - MIN(transaction_date)) as customer_tenure, COUNT(DISTINCT DATE_TRUNC('month', transaction_date)) as active_months FROM transactions GROUP BY customer_id ), behavioral_features AS ( SELECT customer_id, SUM(CASE WHEN transaction_amount > 1000 THEN 1 ELSE 0 END) as high_value_transactions, LAG(transaction_amount, 1) OVER (PARTITION BY customer_id ORDER BY transaction_date) as prev_transaction, ROW_NUMBER() OVER (PARTITION BY customer_id ORDER BY transaction_date DESC) as recency_rank FROM transactions ) SELECT * FROM customer_metrics cm JOIN behavioral_features bf ON cm.customer_id = bf.customer_id;

Statistical Methods Applied

# Python: Advanced Statistical Analysis import scipy.stats as stats import numpy as np from statsmodels.stats.power import ttest_power # Welch's t-test for unequal variances premium_amounts = df[df['account_type'] == 'Premium']['transaction_amount'] standard_amounts = df[df['account_type'] == 'Standard']['transaction_amount'] t_stat, p_value = stats.ttest_ind(premium_amounts, standard_amounts, equal_var=False) effect_size = (premium_amounts.mean() - standard_amounts.mean()) / np.sqrt( (premium_amounts.var() + standard_amounts.var()) / 2 ) # Confidence Intervals confidence_interval = stats.t.interval( confidence=0.95, df=len(premium_amounts)-1, loc=premium_amounts.mean(), scale=stats.sem(premium_amounts) ) print(f"T-statistic: {t_stat:.4f}") print(f"P-value: {p_value:.4f}") print(f"Effect size (Cohen's d): {effect_size:.4f}") print(f"95% CI: [{confidence_interval[0]:.2f}, {confidence_interval[1]:.2f}]")

# Python: Customer Segmentation Model from sklearn.cluster import KMeans from sklearn.preprocessing import StandardScaler from sklearn.decomposition import PCA from sklearn.metrics import silhouette_score # Feature Engineering for Clustering features = ['recency', 'frequency', 'monetary', 'avg_transaction', 'transaction_volatility', 'product_diversity', 'support_tickets', 'login_frequency', 'mobile_usage', 'account_balance'] # Preprocessing scaler = StandardScaler() X_scaled = scaler.fit_transform(df[features]) # PCA for dimensionality reduction pca = PCA(n_components=0.95) # Retain 95% variance X_pca = pca.fit_transform(X_scaled) # Optimal clusters using elbow method and silhouette analysis silhouette_scores = [] for k in range(2, 11): kmeans = KMeans(n_clusters=k, random_state=42, n_init=10) cluster_labels = kmeans.fit_predict(X_pca) silhouette_avg = silhouette_score(X_pca, cluster_labels) silhouette_scores.append(silhouette_avg) # Best model: K=5 clusters optimal_kmeans = KMeans(n_clusters=5, random_state=42, n_init=10) clusters = optimal_kmeans.fit_predict(X_pca)

# Python: Production Model API from fastapi import FastAPI from pydantic import BaseModel import joblib import numpy as np app = FastAPI(title="Churn Prediction API") # Load trained model model = joblib.load('models/churn_prediction_rf.pkl') scaler = joblib.load('models/feature_scaler.pkl') class CustomerFeatures(BaseModel): transaction_frequency: float account_balance: float login_frequency: float support_tickets: int customer_age: int product_count: int avg_transaction: float @app.post("/predict/churn") async def predict_churn(features: CustomerFeatures): # Feature preprocessing feature_array = np.array([[ features.transaction_frequency, features.account_balance, features.login_frequency, features.support_tickets, features.customer_age, features.product_count, features.avg_transaction ]]) # Scale features scaled_features = scaler.transform(feature_array) # Prediction churn_probability = model.predict_proba(scaled_features)[0][1] prediction = model.predict(scaled_features)[0] return { "churn_probability": float(churn_probability), "prediction": "High Risk" if prediction == 1 else "Low Risk", "confidence": float(max(model.predict_proba(scaled_features)[0])) }
