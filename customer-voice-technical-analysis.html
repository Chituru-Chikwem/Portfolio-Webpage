# Topic modeling with coherence optimization from gensim.models import LdaModel from gensim.models.coherencemodel import CoherenceModel class TopicAnalyzer: def __init__(self, num_topics=12): self.num_topics = num_topics self.lda_model = None self.coherence_score = None def train_topic_model(self, processed_texts): # Create dictionary and corpus dictionary = corpora.Dictionary(processed_texts) dictionary.filter_extremes(no_below=5, no_above=0.7) corpus = [dictionary.doc2bow(text) for text in processed_texts] # Train LDA model with hyperparameter tuning self.lda_model = LdaModel( corpus=corpus, id2word=dictionary, num_topics=self.num_topics, alpha='auto', eta='auto', iterations=1000, passes=10, eval_every=10 ) # Calculate coherence score coherence_model = CoherenceModel( model=self.lda_model, texts=processed_texts, dictionary=dictionary, coherence='c_v' ) self.coherence_score = coherence_model.get_coherence() return self.lda_model, self.coherence_score
